##Coalition War - Leader
character_event = {
	id = RTK.130011
	desc = EVTDESC_RTK.130011
	title = EVTTITLE_RTK.130011
	picture = GFX_evt_civil_war
	
	#Fast event triggers
	has_global_flag = coalition_called
	has_character_flag = coalition_leader
	only_playable = yes
	
	trigger = {
		NOT = { has_global_flag = coalition_start }
		prisoner = no
		war = no
		any_independent_ruler = {
			has_character_flag = coalition_target
			prisoner = no
			NOR = {
				war_with = ROOT
				has_truce = ROOT
			}
		}
	}
	
	weight_multiplier = {
		days = 1
		
		modifier = {
			factor = 10
			has_character_flag = coalition_leader
		}
	}
	
	option = { # Start war
		name = "EXCELLENT"
		clr_global_flag = coalition_called
		set_global_flag = coalition_start
		any_independent_ruler = {
			limit = {
				has_character_flag = coalition_target
				NOT = { has_truce = ROOT }
			}
			if = {
				limit = { NOT = { de_jure_liege_or_above = ROOT } }
				primary_title = {
					reverse_war = {
						target = ROOT
						casus_belli = coalition
					}
				}
			}
			if = {
				limit = { de_jure_liege_or_above = ROOT }
				set_defacto_liege = THIS
				primary_title = {
					reverse_war = {
						target = ROOT
						casus_belli = imperial_reconquest
					}
				}
			}
		}
	}
}

##Coalition War - Members
character_event = {
	id = RTK.130012
	desc = EVTDESC_RTK.130012
	title = EVTTITLE_RTK.130012
	picture = GFX_evt_civil_war
	
	#Fast event triggers
	has_global_flag = coalition_start
	has_character_flag = coalition_member
	only_playable = yes
	capable_only = yes
	prisoner = no
	
	trigger = {
		any_independent_ruler = { 
			has_character_flag = coalition_target
			war = yes
			prisoner = no
			is_primary_war_defender = yes
			NOT = { war_with = ROOT }
		}
		any_playable_ruler = {
			has_character_flag = coalition_leader
			war = yes
			is_ruler = yes
			prisoner = no
			is_primary_war_attacker = yes
		}
	}
	
	weight_multiplier = {
		days = 1
		
		modifier = {
			factor = 10
			has_character_flag = coalition_member
		}
	}
	
	option = { # Join war
		name = "EXCELLENT"
		hidden_tooltip = {
			if = {
				limit = {
					top_liege = { has_character_flag = coalition_target }
				}
				set_defacto_liege = THIS
			}
			if = {
				limit = {
					AND = {
						NOT = { top_liege = { has_character_flag = coalition_target } }
						liege = { has_character_flag = coalition_target }
					}
				}
				top_liege = {
					ROOT = { set_defacto_liege = PREV }
				}
			}
		}
		any_playable_ruler = {
			limit = {
				has_character_flag = coalition_leader
			}
			ROOT = {
				join_attacker_wars = PREV
			}
			opinion = {
				modifier = opinion_very_grateful
				who = ROOT
				years = 5
			}
		}
	}
}
