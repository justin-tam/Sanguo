#Spawn Mongols
character_event={
	id=FIN.4
	
	hide_window=yes
	
	is_triggered_only=yes
	
	trigger={
		has_global_flag=wuhu_rise
		NOT={ has_global_flag=wuhu_end }
		NOT={ has_character_flag=wuhu_horde }
		OR={
			k_confucian={
				has_holder=yes
				holder_scope={
					NAND={
						war_with=ROOT
						is_primary_war_defender=yes
					}

				tier=EMPEROR

				}
			}
		}
	}
	
	immediate={
		if={
			limit={ is_tributary=yes }

		}

			limit={
				independent=no
			}
			set_defacto_liege=THIS
		}
		if={
			limit={
				is_nomadic=no
			}
			set_government_type=nomadic_government
		}
		if={
			limit={
				OR={
					culture_group=mongol
					culture=jie
				}
				NOT={ religion=tengri_pagan }
			}
			religion=tengri_pagan
		}
		if={
			limit={
				culture=qiang
				NOT={ religion=baishi }
			}
			religion=baishi
		}
		if={
			limit={
				culture=di
				NOT={ religion=baima }
			}
			religion=baima
		}
		set_character_flag=wuhu_horde
		population=15000
		manpower=5000
		if={
			limit={
				culture_group=mongol
			}
			spawn_mongol_army_effect=yes
		}
		if={
			limit={
				NOT={culture_group=mongol}
			}
			spawn_tibetan_army_effect=yes
		}
		if={
			limit={
				k_confucian={
					has_holder=yes
					holder_scope={
						NAND={
							war_with=ROOT
							is_primary_war_defender=yes
						}
					}
				}
			}
			k_confucian={
				holder_scope={
					save_event_target_as=target_war
				}
			}
		}
		if={
			limit={
				k_confucian={
					has_holder=no
				}
				any_independent_ruler={
					tier=EMPEROR
					realm_size=125
					NAND={
						war_with=ROOT
						is_primary_war_defender=yes
					}
					shares_realm_border_with=ROOT
				}
			}
			random_independent_ruler={
				limit={
					tier=EMPEROR
					realm_size=125
					NAND={
						war_with=ROOT
						is_primary_war_defender=yes
					}
					shares_realm_border_with=ROOT
				}
				save_event_target_as=target_war
			}
		}
		if={
			limit={
				k_confucian={
					has_holder=no
				}
				NOT={
					any_independent_ruler={
						tier=EMPEROR
						realm_size=125
						NAND={
							war_with=ROOT
							is_primary_war_defender=yes
