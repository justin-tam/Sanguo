character_event = {
	id = GRD.2
	
	is_triggered_only = yes
	
	hide_window = yes
	
	immediate = {
		if = { #Check if Regent Chooses
			limit = {
				has_regent = yes
				OR = {
					NOT = { age = 16 }
					prisoner = yes
					trait = incapable
				}
			}
			regent = { character_event = { id = GRD.3 } }
			break = yes
		}
		#Else Emperor chooses
		character_event = { id = GRD.3 }
	}
}

		event_target:target_emperor = {
			any_realm_character = {
				limit = {
					age = 14
					OR = {
						is_female = no
						AND = {
							OR = {
								culture_group = yue
								culture_group = tibetan_group
								culture_group = man
							}
							liege = { religion = ROOT }
						}
						liege = {
							primary_title = {
								has_law = status_of_women_4
							}
						}
					}
					is_married = no
					prisoner = no
					is_ruler = no
					is_primary_heir = no
					NOT = { character = ROOT }
					religion = FROM
					combat_rating = 4
					OR = {
						trait = duelist
						trait = duel_master_warrior
						trait = duel_skilled_warrior
						trait = duel_trained_warrior
					}
					OR = {
						trait = robust

	title = "EVTTITLE_GRD.8"

	picture = "GFX_evt_bodyguard"
	
	trigger = {
		capital_scope = { region = world_china }
		has_landed_title = k_confucian
		is_title_active = c_guard
		c_guard = {
			holder_scope = {
				de_facto_liege = ROOT
				NOT = { has_character_flag = guard_chosen }
			}
		}
		any_courtier = {
			has_character_flag = guard_chosen
			NOT = { has_character_flag = guard_skip }
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		random_courtier = {
			limit = {
				has_character_flag = guard_chosen
				NOT = { has_character_flag = guard_capt }
				NOT = { has_character_flag = guard_skip }
			}
			set_character_flag = guard_capt
		}
	}
	
	option = {
		name = ACCEPT
		
		hidden_tooltip = {
			any_courtier = {
				limit = { has_character_flag = guard_skip }
				clr_character_flag = guard_skip
			}
		}
		
		random_courtier = {
			limit = { has_character_flag = guard_capt }
			hidden_tooltip = { clr_character_flag = guard_capt }
			c_guard = {
				holder_scope = {
					abdicate_to = PREVPREV
					remove_claim = c_guard
